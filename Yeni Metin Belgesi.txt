  <h1>Benimle evlenir misin?</h1>
  <button class="yes" id="yesBtn">Evet</button>
  <button class="no" id="noBtn">Hayır</button>
  <div id="message"></div>

  <script>
    const noBtn = document.getElementById('noBtn');
    const yesBtn = document.getElementById('yesBtn');
    const message = document.getElementById('message');

    const messages = [
      "Evete baş çabuk!",
      "Hala mı uğraşıyorsun?",
      "Bu oyunu kazanamazsın",
      "O buton hep kaçacak",
      "Bence kader evet diyor",
      "Kaçırma şansını",
      "Hala mı uğrasıyorsun",
      "Bence kader evet diyor",
      "Kaçırma şansını",
      "Bence uğraşma evet bas"
    ];

    let attempt = 0;

    function moveButton() {
      if(attempt < messages.length){
        const x = Math.random() * (window.innerWidth - noBtn.offsetWidth);
        const y = Math.random() * (window.innerHeight - noBtn.offsetHeight - 100) + 50;
        noBtn.style.left = x + 'px';
        noBtn.style.top = y + 'px';
        noBtn.style.transform = rotate(${Math.random()*40-20}deg) scale(1.2); // hafif döndür ve büyüt
        setTimeout(() => { noBtn.style.transform = 'rotate(0deg) scale(1)'; }, 300); // normal pozisyona dön
        message.textContent = messages[attempt];
        attempt++;
      } else {
        const x = Math.random() * (window.innerWidth - noBtn.offsetWidth);
        const y = Math.random() * (window.innerHeight - noBtn.offsetHeight - 100) + 50;
        noBtn.style.left = x + 'px';
        noBtn.style.top = y + 'px';
      }
    }

    noBtn.addEventListener('mouseover', moveButton);
    noBtn.addEventListener('click', moveButton);

    yesBtn.addEventListener('click', () => {
      const today = new Date();
      const dateStr = today.toLocaleDateString();
      message.innerHTML = Seni çok seviyorum, iyi ki benimlesin sevgilim 💍<br>Bugünün tarihi: ${dateStr};
    });
  </script>
